Arrays

Basic Operations

Insertion

Time Complexity: O(n) for arbitrary position, O(1) for end


def insert_at_index(arr, index, element):
    if index < 0 or index > len(arr):
        raise IndexError("Index out of bounds")
    arr.append(None)  # Make space
    for i in range(len(arr)-1, index, -1):
        arr[i] = arr[i-1]
    arr[index] = element
    return arr

# Example usage
arr = [1, 2, 3, 4, 5]
insert_at_index(arr, 2, 10)  # Result: [1, 2, 10, 3, 4, 5]

Deletion

Time Complexity: O(n) for arbitrary position, O(1) for end


def delete_at_index(arr, index):
    if index < 0 or index >= len(arr):
        raise IndexError("Index out of bounds")
    for i in range(index, len(arr)-1):
        arr[i] = arr[i+1]
    arr.pop()
    return arr

# Example usage
arr = [1, 2, 3, 4, 5]
delete_at_index(arr, 2)  # Result: [1, 2, 4, 5]

Search

Time Complexity: O(n) for linear search, O(log n) for binary search (sorted array)


def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1  # Element not found

# Example usage
arr = [1, 2, 3, 4, 5]
result = binary_search(arr, 3)  # Result: 2 (index)

Stacks

Stack Operations

Stack Implementation


class Stack:
    def __init__(self):
        self.items = []
        self.min_items = []  # For min stack implementation
    
    def push(self, item):
        self.items.append(item)
        # Update min stack
        if not self.min_items or item <= self.min_items[-1]:
            self.min_items.append(item)
    
    def pop(self):
        if not self.is_empty():
            item = self.items.pop()
            if item == self.min_items[-1]:
                self.min_items.pop()
            return item
        raise IndexError("Pop from empty stack")
    
    def peek(self):
        if not self.is_empty():
            return self.items[-1]
        raise IndexError("Peek at empty stack")
    
    def is_empty(self):
        return len(self.items) == 0
    
    def get_min(self):
        if not self.is_empty():
            return self.min_items[-1]
        raise IndexError("Min from empty stack")

Common Stack Problems

1. Valid Parentheses


def is_valid_parentheses(s):
    stack = []
    brackets = {')': '(', '}': '{', ']': '['}
    
    for char in s:
        if char in brackets.values():
            stack.append(char)
        elif char in brackets:
            if not stack or stack.pop() != brackets[char]:
                return False
    
    return len(stack) == 0

# Example usage
print(is_valid_parentheses("({[]})"))  # True
print(is_valid_parentheses("({[}])"))  # False

Queues

Queue Operations

Queue Implementation


class Queue:
    def __init__(self):
        self.items = []
        self.front = 0
        self.rear = 0
    
    def enqueue(self, item):
        self.items.append(item)
        self.rear += 1
    
    def dequeue(self):
        if not self.is_empty():
            item = self.items[self.front]
            self.front += 1
            if self.front == self.rear:
                self.front = self.rear = 0
                self.items = []
            return item
        raise IndexError("Dequeue from empty queue")
    
    def peek(self):
        if not self.is_empty():
            return self.items[self.front]
        raise IndexError("Peek at empty queue")
    
    def is_empty(self):
        return self.front == self.rear
    
    def size(self):
        return self.rear - self.front

Circular Queue Implementation


class CircularQueue:
    def __init__(self, capacity):
        self.capacity = capacity
        self.items = [None] * capacity
        self.front = self.rear = -1
    
    def enqueue(self, item):
        if self.is_full():
            raise IndexError("Queue is full")
        
        if self.front == -1:
            self.front = 0
        self.rear = (self.rear + 1) % self.capacity
        self.items[self.rear] = item
    
    def dequeue(self):
        if self.is_empty():
            raise IndexError("Queue is empty")
        
        item = self.items[self.front]
        if self.front == self.rear:
            self.front = self.rear = -1
        else:
            self.front = (self.front + 1) % self.capacity
        return item
    
    def is_empty(self):
        return self.front == -1
    
    def is_full(self):
        return (self.rear + 1) % self.capacity == self.front

Graphs

Graph Operations

Graph Implementation


class Graph:
    def __init__(self):
        self.adj_list = {}
    
    def add_node(self, node):
        if node not in self.adj_list:
            self.adj_list[node] = []
    
    def add_edge(self, node1, node2):
        if node1 in self.adj_list and node2 in self.adj_list:
            self.adj_list[node1].append(node2)
            self.adj_list[node2].append(node1)
    
    def remove_node(self, node):
        if node in self.adj_list:
            del self.adj_list[node]
            for neighbors in self.adj_list.values():
                if node in neighbors:
                    neighbors.remove(node)
    
    def remove_edge(self, node1, node2):
        if node1 in self.adj_list and node2 in self.adj_list:
            if node2 in self.adj_list[node1]:
                self.adj_list[node1].remove(node2)
            if node1 in self.adj_list[node2]:
                self.adj_list[node2].remove(node1)

Graph Traversal

1. Breadth-First Search (BFS)


from collections import deque

def bfs(graph, start):
    visited = set()
    queue = deque([start])
    visited.add(start)
    
    while queue:
        node = queue.popleft()
        print(node, end=' ')
        
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

# Example usage
graph = {
    'A': ['B', 'C'],
    'B': ['A', 'D', 'E'],
    'C': ['A', 'F'],
    'D': ['B'],
    'E': ['B', 'F'],
    'F': ['C', 'E']
}
bfs(graph, 'A')

Linked Lists

Introduction to Linked Lists

A linked list is a linear data structure where elements are stored in nodes, and each node points to the next node in the sequence.

Linked List Implementation


class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            return
        current = self.head
        while current.next:
            current = current.next
        current.next = new_node

Trees

Introduction to Trees

A tree is a hierarchical data structure consisting of nodes connected by edges. Each node contains a value and references to other nodes (children).

Binary Tree Implementation


class TreeNode:
    def __init__(self, data):
        self.data = data
        self.left = None
        self.right = None

class BinaryTree:
    def __init__(self):
        self.root = None

    def insert(self, data):
        if not self.root:
            self.root = TreeNode(data)
            return
        
        queue = [self.root]
        while queue:
            node = queue.pop(0)
            if not node.left:
                node.left = TreeNode(data)
                return
            if not node.right:
                node.right = TreeNode(data)
                return
            queue.append(node.left)
            queue.append(node.right)

Sorting Algorithms

Basic Sorting Algorithms

Bubble Sort

Time Complexity: O(n²)


def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

Selection Sort

Time Complexity: O(n²)


def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i+1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

Advanced Sorting Algorithms

Quick Sort

Time Complexity: O(n log n) average case


def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quick_sort(left) + middle + quick_sort(right)

Merge Sort

Time Complexity: O(n log n)


def merge_sort(arr):
    if len(arr) <= 1:
        return arr
        
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    
    return merge(left, right)

def merge(left, right):
    result = []
    i = j = 0
    
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    
    result.extend(left[i:])
    result.extend(right[j:])
    return result

Dynamic Programming

Introduction to Dynamic Programming

Dynamic Programming is a method for solving complex problems by breaking them down into simpler subproblems. It is applicable when subproblems share subsubproblems.

Fibonacci Series using DP


def fibonacci_dp(n):
    # Initialize array to store Fibonacci numbers
    dp = [0] * (n + 1)
    dp[1] = 1
    
    # Build up the solution
    for i in range(2, n + 1):
        dp[i] = dp[i-1] + dp[i-2]
    
    return dp[n]

Common DP Problems

0/1 Knapsack Problem


def knapsack(values, weights, capacity):
    n = len(values)
    dp = [[0 for _ in range(capacity + 1)] for _ in range(n + 1)]
    
    for i in range(1, n + 1):
        for w in range(capacity + 1):
            if weights[i-1] <= w:
                dp[i][w] = max(
                    values[i-1] + dp[i-1][w-weights[i-1]],
                    dp[i-1][w]
                )
            else:
                dp[i][w] = dp[i-1][w]
    
    return dp[n][capacity]

Longest Common Subsequence


def lcs(str1, str2):
    m, n = len(str1), len(str2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if str1[i-1] == str2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    
    return dp[m][n]
